create table Suppliers( MaNhaCC varchar(10) primary key not null, Name nvarchar(255) not null, Address varchar(255) not null, Phone varchar(20) not null, MaSoThue varchar(20) not null unique ); create table TypeServices( MaLoaiDV varchar(10) not null primary key, Name nvarchar(50) not null, ); create table Vehicles( DongXe varchar(10) not null primary key, Name varchar(50) not null, NumberSeat int not null, ); create table FeeItems( MaMP varchar(10) not null primary key, Price decimal(16,0) not null, Decreption nvarchar(255) ); create table Subscriptions( MaDKCC varchar(10) primary key not null, MaNhaCC varchar(10) not null foreign key references Suppliers(MaNhaCC), MaLoaiDV varchar(10) not null foreign key references TypeServices(MaLoaiDV), DongXe varchar(10) not null foreign key references Vehicles(DongXe), MaMP varchar(10) not null foreign key references FeeItems(MaMP), DateStart date not null default GETDATE(), DateStop date not null, Qty int not null default 0, );
